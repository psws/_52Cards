create classes in namespace
	namespace Shiftwise._52cards.mvc.DataEntities
	{
	 public class Rule
		{
			[Key]
			[Column(Order = 1, TypeName = "varchar")]
			[MaxLength(20)]
			[Required]
			public string DeckId { get; set; }
			[Key]
			[Column(Order = 2, TypeName = "varchar")]
			[MaxLength(35)]
			[Required]
			public string GameName { get; set; }
			public short value { get; set; }
			//FK
			public virtual Deck Deck { get; set; }

		}
	}

setup configuration string in web.config
    <add name="_52CardSQLExpress2012" connectionString="Data Source=(LocalDb)\v11.0;AttachDbFilename=|DataDirectory|\_52Cards.mdf;Integrated Security=True" providerName="System.Data.SqlClient" />
    <add name="_52CardSQLExpress2014" connectionString="Data Source=(localdb)\MSSQLLocalDB;AttachDbFilename=|DataDirectory|\_52Cards.mdf;Integrated Security=True" providerName="System.Data.SqlClient" />

Then run this command in the pm
PM> Enable-Migrations -Force -ProjectName Shiftwise._52cards.mvc.DataModel -StartupProjectName Shiftwise._52cards.mvc.App -ContextTypeName Shiftwise._52cards.mvc.DataModel._52CardsDB -MigrationsDirectory Migrations\_52CardsDB

Next do the Add-Migration

PM> Add-Migration -ProjectName Shiftwise._52cards.mvc.DataModel -Force -StartupProjectName Shiftwise._52cards.mvc.App -ConfigurationTypeName Shiftwise._52cards.mvc.DataModel.Migrations._52CardsDB.Configuration "InitialDatabaseCreation"
